1. Update `internal/connection/dial.go` `handleConn`:
   - After `wg.Wait()` returns, check if shutdown was triggered (m.stopCh closed):
   ```go
   select {
   case <-m.stopCh:
       // Clean shutdown — send disconnect frame before closing
       shutCtx, shutCancel := context.WithTimeout(context.Background(), 5*time.Second)
       defer shutCancel()
       disconnectEnv, err := protocol.Encode(protocol.TypeNodeDisconnect, generateMsgID(), protocol.NodeDisconnect{Reason: "shutdown"})
       if err == nil {
           data, _ := json.Marshal(disconnectEnv)
           _ = conn.Write(shutCtx, websocket.MessageText, data)
       }
       conn.Close(websocket.StatusNormalClosure, "shutdown")
       return
   default:
       // Connection dropped — will reconnect via dialLoop
       return
   }
   ```
   - NOTE: The disconnect frame is written directly to conn (not via sendCh), because the writer goroutine has already exited at this point (wg.Wait returned).

2. Update `internal/connection/dial.go` `dialLoop`:
   - After `handleConn` returns, before looping back to reconnect, check `m.stopCh`:
   ```go
   select {
   case <-ctx.Done():
       close(m.stopped)
       return
   case <-m.stopCh:
       close(m.stopped)
       return
   default:
       // reconnect
   }
   ```
   - Add backoff delay between reconnect attempts:
   ```go
   delay := backoff.Next()
   m.log.Info().Dur("retry_in", delay).Msg("reconnecting")
   select {
   case <-time.After(delay):
   case <-ctx.Done():
       close(m.stopped)
       return
   }
   ```
